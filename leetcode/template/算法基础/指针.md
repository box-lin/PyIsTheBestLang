
## 快慢指针
```Python3
def circle(head):
    fast = slow = head
    while fast and fast.next:
        fast = fast.next.next
        slow = slow.next
        if fast==slow:
            return True
    return False
```
## 双指针
```Python3
class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        pre = set()
        j = 0
        ans = 0
        cur = 0
        for i in range(n):
            while j<n and nums[j] not in pre:
                pre.add(nums[j])
                cur += nums[j]
                j += 1
            
            if cur > ans:
                ans = cur
            
            pre.discard(nums[i])
            cur -= nums[i]
        return ans

```
