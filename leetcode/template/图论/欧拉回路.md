class Solution:
    def validArrangement(self, pairs: List[List[int]]) -> List[List[int]]:
        # 存顶点的出入度
        degree = defaultdict(lambda: [0, 0])
        # 存储图关系
        edge = defaultdict(list)
        for i, j in pairs:
            degree[i][0] += 1
            degree[j][1] += 1
            edge[i].append(j)

        # 寻找起始节点
        start = pairs[0][0]
        for i in degree:
            # 如果有节点出度比入度恰好多 1，那么只有它才能是起始节点
            if degree[i][0] - degree[i][1] == 1:
                start = i
                break

        def dfs(pre):
            # 深度优先搜索（Hierholzer 算法）求解欧拉通路
            while edge[pre]:
                nex = edge[pre].pop()
                dfs(nex)
                ans.append([pre, nex])
            return

        ans = list()
        dfs(start)
        return ans[::-1]
