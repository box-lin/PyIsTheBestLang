class Trie:
    def __init__(self):
        self.dct = dict()
        return

    def update(self, word):
        cur = self.dct
        for w in word:
            if w not in cur:
                cur[w] = dict()
            cur = cur[w]
        cur['isEnd'] = 1
        return

    def query(self, word):
        cur = self.dct
        for w in word:
            if w not in cur:
                return False
            cur = cur[w]
            if 'isEnd' not in cur:
                return False
        return True
