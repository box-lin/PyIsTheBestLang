```python3
class Solution:
    def lexicalOrder(self, n: int) -> List[int]:

        def dfs():
            nonlocal ans, pre
            if pre <= n:
                ans.append(pre)
            else:
                return
            for x in range(10):
                pre = pre * 10 + x
                dfs()
                pre = (pre - x) // 10
            return

        ans = []
        for i in range(1, 10):
            pre = i
            dfs()
        return ans
```

```python3
class Solution:
    def lexicalOrder(self, n: int) -> List[int]:
        ans = [0] * n
        num = 1
        for i in range(n):
            ans[i] = num
            if num * 10 <= n:
                # 优先枚举更多位
                num *= 10
            else:
                # 如果超限则回退一位再加一
                while num % 10 == 9 or num + 1 > n:
                    num //= 10
                num += 1
        return ans
```
